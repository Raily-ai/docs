---
title: "List Content"
api: "GET /v1/content"
description: "Retrieve a list of all content items"
---

## Request

<ParamField query="limit" type="integer" default="20">
  Number of items to return (1-100)
</ParamField>

<ParamField query="cursor" type="string">
  Pagination cursor from previous response
</ParamField>

<ParamField query="type" type="string">
  Filter by content type: `article`, `report`, `image`, `dataset`, `video`
</ParamField>

<ParamField query="policyId" type="string">
  Filter by policy ID
</ParamField>

<ParamField query="created[gte]" type="string">
  Filter by creation date (ISO 8601)
</ParamField>

<ParamField query="created[lte]" type="string">
  Filter by creation date (ISO 8601)
</ParamField>

<ParamField query="expand" type="string">
  Expand related objects: `policy`
</ParamField>

## Response

<ResponseField name="object" type="string">
  Always `list`
</ResponseField>

<ResponseField name="data" type="array">
  Array of content objects
  <Expandable title="Content Object">
    <ResponseField name="id" type="string">
      Unique identifier (e.g., `cnt_abc123`)
    </ResponseField>
    <ResponseField name="object" type="string">
      Always `content`
    </ResponseField>
    <ResponseField name="externalId" type="string">
      Your external identifier
    </ResponseField>
    <ResponseField name="title" type="string">
      Content title
    </ResponseField>
    <ResponseField name="type" type="string">
      Content type
    </ResponseField>
    <ResponseField name="source" type="string">
      Source URL
    </ResponseField>
    <ResponseField name="policyId" type="string">
      Applied policy ID
    </ResponseField>
    <ResponseField name="metadata" type="object">
      Custom metadata
    </ResponseField>
    <ResponseField name="created" type="string">
      ISO 8601 timestamp
    </ResponseField>
    <ResponseField name="updated" type="string">
      ISO 8601 timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="hasMore" type="boolean">
  Whether more items exist
</ResponseField>

<ResponseField name="nextCursor" type="string">
  Cursor for next page
</ResponseField>

<ResponseField name="totalCount" type="integer">
  Total number of matching items
</ResponseField>

<RequestExample>

```bash cURL
curl https://api.raily.ai/v1/content?limit=10&type=article \
  -H "Authorization: Bearer raily_sk_xxxxx"
```

```javascript JavaScript
const content = await raily.content.list({
  limit: 10,
  type: 'article'
});

console.log(`Found ${content.totalCount} articles`);
content.data.forEach(item => {
  console.log(`${item.title} (${item.id})`);
});
```

```python Python
content = raily.content.list(
    limit=10,
    type="article"
)

print(f"Found {content.total_count} articles")
for item in content.data:
    print(f"{item.title} ({item.id})")
```

</RequestExample>
